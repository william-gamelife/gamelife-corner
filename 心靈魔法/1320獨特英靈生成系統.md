// 動態生成描述
function generateUniqueSpirit(first, second, third, allScores) {
  // 1. 計算分數差距
  const gap1 = allScores[first] - allScores[second];  // 主次差距
  const gap2 = allScores[second] - allScores[third];  // 次隱差距
  const balance = calculateBalance(allScores);        // 整體平衡度
  
  // 2. 根據差距調整描述
  const personality = generatePersonality(first, second, third, gap1, gap2);
  
  // 3. 加入獨特元素
  const uniqueTraits = generateUniqueTraits(first, second, third, balance);
  
  return combineAll(personality, uniqueTraits);
}

## 五、實作範例

### 5.1 完整的生成流程

```javascript
class UniqueSpritGenerator {
  
  generate(testResult) {
    // 1. 解析測驗結果
    const [first, second, third] = testResult.topThree;
    const scores = testResult.scores;
    
    // 2. 計算獨特參數
    const params = {
      combo: [first, second, third],
      gaps: this.calculateGaps(scores),
      balance: this.calculateBalance(scores),
      weakest: this.findWeakest(scores),
      strengths: this.analyzeStrengths(scores)
    };
    
    // 3. 生成獨特內容
    const spirit = {
      // 編號（1-1320）
      id: this.getComboId(params.combo),
      
      // 獨特稱號
      title: this.generateTitle(params),
      
      // 個性本質（獨特描述）
      personality: this.generatePersonality(params),
      
      // 三大優勢（根據分數差距調整）
      advantages: this.generateAdvantages(params),
      
      // 成長建議（基於最弱的神）
      growth: this.generateGrowth(params),
      
      // 靈魂詩籤（獨特的詩）
      poem: this.generatePoem(params),
      
      // 稀有度
      rarity: this.calculateRarity(params.combo)
    };
    
    return spirit;
  }
  
  // 確保1320個都不同
  generatePersonality(params) {
    const [p, s, h] = params.combo;
    
    // 每個組合都有專屬模板
    const template = this.templates[`${p}-${s}-${h}`];
    
    // 根據分數差距微調
    const adjusted = this.adjustByGaps(template, params.gaps);
    
    // 加入個人化元素
    const personalized = this.personalize(adjusted, params);
    
    return personalized;
  }
}
```

## 八、驗證1320個都不同的方法

```javascript
// 測試腳本
function validateUniqueness() {
  const allCombos = generateAllCombinations(); // 1320個
  const generated = new Set();
  
  for(let combo of allCombos) {
    const result = generator.generate(combo);
    const key = result.personality + result.title;
    
    if(generated.has(key)) {
      console.error(`重複了！${combo}`);
    }
    generated.add(key);
  }
  
  console.log(`成功生成${generated.size}個獨特英靈`);
}
```

這樣每個人測完都會得到專屬於那個組合的獨特描述！
