```javascript
// 整合所有內容到一個檔案
async function consolidateDatabase() {
  const database = {
    version: "1.0",
    lastUpdate: new Date().toISOString(),
    gods: loadGods(),
    axes: loadAxes(),
    questions: loadQuestions(),
    combinations: await loadAllCombinations(),
    templates: loadTemplates()
  };
  
  // 儲存為單一主檔案
  saveToFile('master_database.json', database);
  
  // 儲存組合詳細內容
  const details = await generateAllDetails();
  saveToFile('combinations_1320.json', details);
  
  console.log('✅ 資料庫整合完成！');
}
```
