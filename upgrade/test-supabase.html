4. 建立資料表 SQL

如果表不存在，請在 Supabase SQL Editor 中執行以下 SQL：

-- 建立 user_data 表
CREATE TABLE IF NOT EXISTS user_data (
    id BIGSERIAL PRIMARY KEY,
    user_id UUID NOT NULL,
    module TEXT NOT NULL,
    data JSONB NOT NULL,
    created_at TIMESTAMPTZ DEFAULT now(),
    updated_at TIMESTAMPTZ DEFAULT now(),
    CONSTRAINT user_data_unique UNIQUE (user_id, module)
);

-- 開啟 RLS (Row Level Security)
ALTER TABLE user_data ENABLE ROW LEVEL SECURITY;

-- 建立政策允許所有人讀寫資料 (開發環境用)
CREATE POLICY "Allow all access for development" ON user_data
    FOR ALL 
    USING (true)
    WITH CHECK (true);

-- 建立索引提升性能
CREATE INDEX IF NOT EXISTS idx_user_data_user_id ON user_data(user_id);
CREATE INDEX IF NOT EXISTS idx_user_data_module ON user_data(module);
